Web Mail

Disable SELINUX

[root@localhost ~]# nano /etc/selinux/config
set SELINUX=disabled
[root@localhost ~]# reboot

Install & Konfigurasi Postfix

[root@localhost ~]# yum install epel-release postfix -y
[root@localhost ~]# nano /etc/postfix/main.cf
Edit pada baris berikut
## Enter your hostname
myhostname = osradar.localhost.local

## Enter your domain name
mydomain = osradar.localhost.local

inet_interfaces = all
myorigin = $mydomain

##set your IP range
mynetworks = 192.168.130.138/24, 127.0.0.0/8

## set your mail direcotry
home_mailbox = Maildir/

## Now Go to line no 164 add $mydomain at the end
mydestination = $myhostname, localhost.$mydomain, localhost, $mydomain
Simpan & keluar

[root@localhost ~]# systemctl restart postfix
[root@localhost ~]# systemctl enable postfix

Install & Konfigurasi Dovecot

[root@localhost ~]# yum install dovecot -y
[root@localhost ~]# systemctl start dovecot
[root@localhost ~]# systemctl enable dovecot
[root@localhost ~]# nano /etc/dovecot/dovecot.conf
Cari dan ubah kode sesuai ini
protocols = imap pop3 lmtp

[root@localhost ~]# nano /etc/dovecot/conf.d/10-mail.conf
Cari dan ubah kode sesuai ini
mail_location = maildir:~/Maildir

[root@localhost ~]# nano /etc/dovecot/conf.d/10-auth.conf
Cari dan ubah kode sesuai ini
disable_plaintext_auth = yes
auth_mechanisms = plain login

[root@localhost ~]# nano /etc/dovecot/conf.d/10-master.conf
Cari dan ubah kode sesuai ini
unix_listener auth-userdb {
#mode = 0600
user = postfix delete ## on line 91
group = postfix delete ## on line 92

[root@localhost ~]# systemctl restart dovecot
[root@localhost ~]# systemctl enable dovecot

Install & Konfigurasi Squirrelmail

[root@localhost ~]# yum install httpd squirrelmail -y
[root@localhost ~]# systemctl start httpd
[root@localhost ~]# systemctl enable httpd
[root@localhost ~]# cd /usr/share/squirrelmail/config/
[root@localhost ~]# ./conf.pl
Langsung keluar saja

[root@localhost ~]# nano /etc/httpd/conf/httpd.conf
Tambahkan pada file httpd.conf
Alias /webmail /usr/share/squirrelmail
<Directory /usr/share/squirrelmail>
Options Indexes FollowSymLinks
RewriteEngine On
AllowOverride All
DirectoryIndex index.php
Order allow,deny
Allow from all
</Directory>
Simpan dan keluar

[root@localhost ~]# systemctl restart httpd
[root@localhost ~]# systemctl restart dovecot
