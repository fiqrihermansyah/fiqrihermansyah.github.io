Mail Server

Install & Konfigurasi Postfix

[root@localhost ~]# yum install postfix -y
[root@localhost ~]# nano /etc/postfix/main.cf
Menuju baris 83 pada kata " mydomain = " hapus tanda pagar dan masukan domain kita
Menuju baris 75 pada kata " myhostname = " hapus tanda pagar dan masukan domain dari mail server
Menuju baris 99 pada kata " myorigin = " hapus tanda pagar
Menuju baris 116 pada kata " inet_interfaces = " hapus tanda pagar dan ganti localhost menjadi all
Menuju baris 164 pada kata " mydestination = " hapus tanda pagar dan tambahkan mydestination
Menuju baris 264 pada kata " mynetworks = " hapus tanda pagar dan tambahkan IP centOS
Menuju baris 419 pada kata " home_mailbox " hapus tanda pagar dan isikan Maildir/
Menuju baris 574 pada kata " smptd_banner " hapus tanda pagar dan isikan $myhostname ESMTP
Menuju baris paling bawah dan isikan skrip berikut
message_size_limit = 10485760
mailbox_size_limit = 1073741824

smtpd_sasl_type = dovecot
smtpd_sasl_path = private/auth
smtpd_sasl_auth_enable = yes
smtpd_sasl_security_options = noanonymous
smtpd_sasl_local_domain = $myhostname
smtpd_recipient_restrictions = permit_mynetworks,permit_auth_destination,permit_sasl_authenticated,reject

[root@localhost ~]# systemctl restart postfix
[root@localhost ~]# systemctl enable postfix
[root@localhost ~]# firewall-cmd --add-service=smtp --permanent
[root@localhost ~]# firewall-cmd --reload

Install & Konfigurasi Dovecot

[root@localhost ~]# yum install dovecot -y
[root@localhost ~]# nano /etc/dovecot/dovecot.conf
Menuju baris 24 pada kata " protocols " hapus tanda pagar dan pastikan isinya imap pop3 lmtp
Menuju baris 30 pada kata " listen " hapus tanda pagar dan hanya kita isikan *

[root@localhost ~]# nano /etc/dovecot/conf.d/10-auth.conf
Menuju baris 10 pada kata " disable_plaintext_auth " hapus tanda pagar dan ganti dari yes menjadi no
Menuju baris 100 pada kata " auth_mechanism " hapus tanda pagar dan tambahkan kata login dibelakang plain

[root@localhost ~]# nano /etc/dovecot/conf.d/10-mail.conf
Menuju baris 30 pada kata " mail_location " hapus tanda pagar dan kita isi dengan maildir:~/Maildir

[root@localhost ~]# nano /etc/dovecot/conf.d/10-master.conf
Menuju baris 96 pada kata " mail_location " hapus tanda pagar pada unix sampai } dan kita edit menjadi
unix_listener /var/spool/postfix/private/auth {
               mode = 0666
               user = postfix
               group = postfix
}

[root@localhost ~]# nano /etc/dovecot/conf.d/10-ssl.conf
Menuju baris 8 pada kata " ssl = " hapus tanda pagar dan kita isi dengan no

[root@localhost ~]# systemctl restart dovecot
[root@localhost ~]# systemctl enable dovecot
[root@localhost ~]# firewall-cmd --add-port={110/tcp,143/tcp} --permanent
[root@localhost ~]# firewall-cmd --reload

Menambah User

[root@localhost ~]# useradd namauser

Pengujian (Pengiriman)

[root@localhost ~]# telnet mail.fiqri.mkt 25
[root@localhost ~]# halo fiqri
[root@localhost ~]# mail from: fiqri
[root@localhost ~]# rcpt to: herman
[root@localhost ~]# data
[root@localhost ~]# Isi pesan tulis disini jika pesan sudah diisi, lalu enter dan enter lagi kemudian tutup dengan tanda titik (.)

Pengujian (Penerimaan)

[root@localhost ~]# telnet mail.fiqri.mkt 110
[root@localhost ~]# user herman
[root@localhost ~]# pass herman
[root@localhost ~]# stat
[root@localhost ~]# retr 1 berfungsi membuka email menurut nomer yang diinput
[root@localhost ~]# quit
